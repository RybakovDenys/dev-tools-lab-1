21def bounding_box(cells: set[tuple[int, int]]) -> tuple[int, int, int, int] | None:
22 """!
23 @brief Compute minimal axis-aligned bounding box of live cells.
24 @param cells Set of (x,y) integer tuples.
25 @return (min_x, min_y, max_x, max_y) tuple or None if set empty.
26 @details Iterates once over coordinates collecting min/max extents.
27 @note Returns None for an empty input set to distinguish from a degenerate box.
28 @example Basic usage
29 box = bounding_box({(2,1),(4,3),(3,2)}) # (2,1,4,3)
30 @example Empty set
31 assert bounding_box(set()) is None
32 """
33 if not cells:
34 return None
35 xs = [x for x, _ in cells]
36 ys = [y for _, y in cells]
37 return (min(xs), min(ys), max(xs), max(ys))

39def coords(*pairs):
40 """!
41 @brief Convenience helper to build a set of (x,y) coordinate tuples.
42 @param pairs Variadic sequence of (x,y) tuples.
43 @return Set of provided tuples.
44 @details Accepts any number of (x,y) tuples and returns them as a set; duplicates collapse.
45 @example
46 live = coords((1,1), (2,2), (1,1)) # {(1,1), (2,2)}
47 """
48 return set(pairs)

50def translate(cells: set[tuple[int, int]], dx: int, dy: int, width: int | None = None, height: int | None = None) -> set[tuple[int, int]]:
51 """!
52 @brief Translate a set of cells by (dx, dy), optionally wrapping.
53 @param cells Set of (x,y) integer tuples.
54 @param dx Horizontal shift.
55 @param dy Vertical shift.
56 @param width Optional wrap width.
57 @param height Optional wrap height (required if width provided).
58 @return New set of translated (and possibly wrapped) cells.
59 @details If both width and height are provided, wrapping is applied modulo
60 those dimensions; otherwise translation is absolute.
61 @example Absolute translation
62 translate({(1,1),(2,2)}, 3, -1) # {(4,0),(5,1)}
63 @example Toroidal translation
64 translate({(4,4)}, 1, 1, width=5, height=5) # {(0,0)}
65 """
66 result: set[tuple[int, int]] = set()
67 for x, y in cells:
68 nx, ny = x + dx, y + dy
69 if width is not None and height is not None:
70 nx %= width
71 ny %= height
72 result.add((nx, ny))
73 return result

75def canonicalize(cells: set[tuple[int, int]]) -> set[tuple[int, int]]:
76 """!
77 @brief Normalize a shape so its top-left cell becomes (0,0).
78 @param cells Set of (x,y) tuples.
79 @return New set with all coordinates shifted.
80 @details Useful for comparing pattern shapes independent of translation. Empty input yields empty set.
81 @example Basic normalization
82 canonicalize({(2,3),(3,3),(2,4)}) # {(0,0),(1,0),(0,1)}
83 @example Idempotence
84 s = {(0,0),(1,0)}; assert canonicalize(s) == s
85 """
86 if not cells:
87 return set()
88 min_x = min(x for x, _ in cells)
89 min_y = min(y for _, y in cells)
90 return {(x - min_x, y - min_y) for x, y in cells}
